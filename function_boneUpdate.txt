function boneUpdate() {
	socket.on('updateBones', function(data) {
		for(var boneData in data) {

		// update array of bones
		var bone = {};
		bone.name = data[boneData].name;
		bone.posX = data[boneData].pos[0];
		bone.posY = data[boneData].pos[1];
		bone.posZ = data[boneData].pos[2];
		bone.quatX = data[boneData].rot[1];
		bone.quatY = data[boneData].rot[2];
		bone.quatZ = data[boneData].rot[3];
		bone.quatW = data[boneData].rot[0];

		bones[bone.name] = bone;
		}
	});

	// loop through all bones
	for(var boneData in bones) {

	brad.skeleton.bones[boneData.name].quaternion.set(boneData.quatX,boneData.quatY,boneData.quatZ,boneData.quatW);
	brad.skeleton.bones[boneData.name].position.set(boneData.posX,boneData.posY,boneData.posZ);
	}
}
